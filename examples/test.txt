def mul a b
begin
    ifzero b
        begin
            return 0
        end
    else
        begin
            let newb
            give newb inc b -1

            let next_mul
            give next_mul mul a newb
            
            return inc a next_mul
        end
end

def fat n
begin
    ifzero n
    begin
        return 1
    end
    let new_n
    give new_n inc n -1
    let res
    give res fat new_n
    give res mul res n
    return res
end

def soma a b
begin
    return inc a b
end

def main
begin
    return fat 4
end
